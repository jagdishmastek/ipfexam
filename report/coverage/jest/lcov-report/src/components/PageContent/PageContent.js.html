import React from 'react'
import { Link, useLocation } from 'react-router-dom';
import { FontAwesomeIcon } from '@fortawesome/react-fontawesome';
import { faFaceFrown, faTriangleExclamation } from '@fortawesome/free-solid-svg-icons';
import Header from '../Header/Header'
import "./PageContent.css"
 
const PageContent = (props) => {
    let location = useLocation();
    let path = location.pathname.replace("/","");
    let title = "Page Not Found";   
    let contentText = "The page you looking for dosen't exist.";
    let pageImage = faFaceFrown; 
    if(path === "error-page"){
        title =  "Something went wrong";
        contentText = "Something went wrong" 
        pageImage = faTriangleExclamation; 
    }
 
    return (
        <div className='page-content'>
            <Header title={title} />
            <div className="pagecontent-container">
                
            <div className="container">
                <div className="row">
                    <div className="col-md-12">
                        <div className='pagecontent-div'> 
                            <div>
                                <h2>{contentText}</h2>
                            </div>
                            <div>
                                <FontAwesomeIcon icon={pageImage} 
                                style={{ color: 'grey', height:'200px' }}/>
                            </div>
                        </div>
                    </div>
                </div>
                </div>
                <div className="pagecontent-footer">
                    <Link to="/">Go Back</Link>
                </div>
            </div>
            
        </div>
    )
}
 
export default PageContent;
